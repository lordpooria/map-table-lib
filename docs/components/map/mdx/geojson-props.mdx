## GEO json props

These are available geojson props you could pass to time-dimension component

You can Customize GeoJSON layer props as well, Right now we just has polyline props, circle props and path options.

import GeoJSONProps from "../components/GeoJSONProps";

<GeoJSONProps />

import SyntaxHighlighter from "../../SyntaxHighlighter";

<SyntaxHighlighter>
  {`import React from "react";
import HesabaTimeDimension from "@hesaba/map";
import data from "../data/data.json";
export default function BaseMapScreen() {
 return(
  return (
    <HesabaTimeDimension
      mapProps={{
        center: [35.76498031616211, 51.33673858642578],
        zoom: 13,
        zoomControl: false,
      }}
      data={data}
      layerProps={{
        circleProps: { radius: 10 },
        pathOptions: { fillColor: "#0FF", color: "#F0F" },
        polylineProps: { opacity: 0.4 },
      }}
    />
  );
 )
}`}
</SyntaxHighlighter>
;
