## Custom Font Family ClassName

You could add font family like this with className approach

**_but you should consider !important in styels_**

import CustomFontFamily from "../components/CustomFontFamily";

<CustomFontFamily />

import SyntaxHighlighter from "../../SyntaxHighlighter";

<SyntaxHighlighter>
  {`import React from "react";
import HesabaTimeDimension from "@hesaba/map";
import data from "../data/data.json";
import { makeStyles } from "@material-ui/core";
const useStyles = makeStyles({
  container: { fontFamily: "Vazir !important" },
});
export default function BaseMapScreen() {
 return(
  return (
    <HesabaTimeDimension
      mapProps={{
        center: [35.76498031616211, 51.33673858642578],
        zoom: 13,
        zoomControl: false,
        className: classes.container
      }}
      data={data}
    />
  );
 )
}
`}
</SyntaxHighlighter>
;

Or you could simply use vanilla css styles

<SyntaxHighlighter>
  {`
import React from "react";
import HesabaTimeDimension from "@hesaba/map";
import data from "../data/data.json";
import "./css/map.css";
export default function BaseMapScreen() {
 return(
  return (
    <HesabaTimeDimension
      mapProps={{
        center: [35.76498031616211, 51.33673858642578],
        zoom: 13,
        zoomControl: false,
      }}
      data={data as any}
    />
  );
 )
}
`}
</SyntaxHighlighter>
;

You could add map.css like this to push your font family to the map component:

<SyntaxHighlighter>
  {`
.leaflet-container {
  font-family: Vazir;
}
`}
</SyntaxHighlighter>
;

You also could add this global inside you app all.css(or global app css) files
