## Map with Table

This component can be rendered with a table beside it, Table contains all properties of geojson provided to 
time dimension, simply by adding withTable attribute to it

import MapWithTable from "../components/MapWithTable";

<MapWithTable />

import SyntaxHighlighter from "../../SyntaxHighlighter";

<SyntaxHighlighter>
  {`import React from "react";
import HesabaTimeDimension from "@hesaba/map";
import data from "../data/data.json";
export default function BaseMapScreen() {
 return(
  return (
    <HesabaTimeDimension
      mapProps={{
        center: [35.76498031616211, 51.33673858642578],
        zoom: 13,
        zoomControl: false,
      }}
      data={data}
      withTable
    />
  );
 )
}`}
</SyntaxHighlighter>
;
