## Custom Style Map Different Player Thumb

You could simply change player theme by passing a material theme object to it

import { MaterialThemeObject } from "../../components/CustomStyles";

<MaterialThemeObject />

import SyntaxHighlighter from "../../../SyntaxHighlighter";

<SyntaxHighlighter>
  {`
import data from "../assets/small_data.json";
import HesabaTimeDimension from "@hesaba/map";
import { makeStyles } from "@material-ui/core";
import theme from "../styles/theme";
export default function MapComponent(props: any) {
  const classes = useStyles();
  return (
    <HesabaTimeDimension
      classes={{ tdRoot: classes.root }}
      mapProps={{
        center: [35.76498031616211, 51.33673858642578],
        zoom: 13,
        zoomControl: false,
        className: classes.map,
      }}
      data={data}
      theme={theme}
    />
  );
}
    `}
</SyntaxHighlighter>
;
