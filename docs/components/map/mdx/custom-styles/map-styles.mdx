## Custom Style For map component

obviously you could change map styls and it's root container styles

import BaseMap from "../../components/BaseMap";

<BaseMap />

import SyntaxHighlighter from "../../../SyntaxHighlighter";

<SyntaxHighlighter>
  {`
import data from "../assets/small_data.json";
import HesabaTimeDimension from "@hesaba/map";
import { makeStyles } from "@material-ui/core";
const useStyles = makeStyles({
  root: { width: "70vw", height: "70vh" },
  map: { width: "100%", height: "100%" },
});
export default function MapComponent(props: any) {
  const classes = useStyles();
  return (
    <HesabaTimeDimension
      classes={{ tdRoot: classes.root }}
      mapProps={{
        center: [35.76498031616211, 51.33673858642578],
        zoom: 13,
        zoomControl: false,
        className: classes.map,
      }}
      data={data}
    />
  );
}
    `}
</SyntaxHighlighter>
;
